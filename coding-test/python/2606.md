cpu = int(input())
connect = int(input())

cpus = [set() for _ in range(cpu + 1)]
for _ in range(connect):
    pc1, pc2 = map(int, input().split())
    cpus[pc1].update([pc1, pc2])
    cpus[pc2].update([pc1, pc2])
    
visited = [False] * (cpu + 1)
def dfs(x):
    visited[x] = True
    count = 1
    for pc in cpus[x]:
        if not visited[pc]:
            count += dfs(pc)
    return count
    
print(dfs(1) - 1)
